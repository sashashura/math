71  using scalar_t = return_type_t<Ta, Tb, Tz>;
72  using Ta_plain = plain_type_t<Ta>;
73  using Tb_plain = plain_type_t<Tb>;
74  using T_vec = Eigen::Matrix<scalar_t, -1, 1>;
75  ref_type_t<Ta> a_ref_in = a;
76  ref_type_t<Tb> b_ref_in = b;
77 
78  // Replace any zero inputs with the smallest number representable, so that
79  // taking log and aggregating does not return -Inf
80  Ta_plain a_ref
81  = (a_ref_in.array() == 0).select(EPSILON, a_ref_in.array()).matrix();
82  Tb_plain b_ref
83  = (b_ref_in.array() == 0).select(EPSILON, b_ref_in.array()).matrix();
84  int a_size = a.size();
85  int b_size = b.size();
86 
87  // Convergence criteria for the gradients are the same as for the
88  // Hypergeometric pFq itself
89  bool condition_1 = (a_size > (b_size + 1)) && (z != 0);
90  bool condition_2 = (a_size == (b_size + 1)) && (fabs(z) > 1);
91 
92  if (condition_1 || condition_2) {
93  std::stringstream msg;
94  msg << "hypergeometric pFq gradient does not meet convergence "
95  << "conditions with given arguments. "
96  << "a: " << a_ref << ", b: " << b_ref << ", "
97  << "z: " << z;
98  throw std::domain_error(msg.str());
99  }
100 
101  // As the gradients will be aggregating on the log scale, we will track the
102  // the values and the signs separately - to avoid taking the log of a
103  // negative input
104  Eigen::VectorXi a_signs = sign(value_of_rec(a_ref));
105  Eigen::VectorXi b_signs = sign(value_of_rec(b_ref));
106  int z_sign = sign(value_of_rec(z));
107 
108  Ta_plain ap1 = (a.array() + 1).matrix();
109  Tb_plain bp1 = (b.array() + 1).matrix();
110  scalar_type_t<Tz> log_z = log(fabs(z));
111  scalar_type_t<Ta> a_prod = prod(a_ref);
112  scalar_type_t<Tb> b_prod = prod(b_ref);
113 
114  // Only need the infinite sum for partials wrt a & b
115  if (calc_a || calc_b) {
116  double log_precision = log(precision);
117 
118  T_vec da_mn = T_vec::Constant(a_size, NEGATIVE_INFTY);
119  T_vec db_mn = T_vec::Constant(b_size, NEGATIVE_INFTY);
120  T_vec da = T_vec::Constant(a_size, 0.0);
121  T_vec db = T_vec::Constant(b_size, 0.0);
122 
123  int m = 0;
124  int n_iter = 2;
125 
126  double lgamma_mp1 = 0;
127  double log_phammer_1m = 0;
128  double log_phammer_2m = 0;
129  T_vec log_phammer_ap1_m = T_vec::Zero(ap1.size());
130  T_vec log_phammer_bp1_m = T_vec::Zero(bp1.size());
131  Tz log_z_m = 0;
132  Ta_plain ap1m = ap1;
133  Tb_plain bp1m = bp1;
134 
135  Ta_plain ap1n = ap1;
136  Tb_plain bp1n = bp1;
137  Ta_plain an = a_ref;
138  Tb_plain bn = b_ref;
139  Ta_plain ap1mn = ap1m;
140  Tb_plain bp1mn = bp1m;
141 
142  double log_phammer_1n;
143  double log_phammer_2_mpn;
144  double lgamma_np1;
145  T_vec log_phammer_an(a_size);
146  T_vec log_phammer_ap1_n(a_size);
147  T_vec log_phammer_bp1_n(b_size);
148  T_vec log_phammer_bn(b_size);
149  T_vec log_phammer_ap1_mpn(a_size);
150  T_vec log_phammer_bp1_mpn(b_size);
151 
152  int z_pow_m_sign = 1;
153  Eigen::VectorXi curr_signs_da(a_size);
154  Eigen::VectorXi curr_signs_db(b_size);
155  Eigen::VectorXi log_phammer_an_sign(a_size);
156  Eigen::VectorXi log_phammer_ap1n_sign(a_size);
157  Eigen::VectorXi log_phammer_bp1n_sign(b_size);
158  Eigen::VectorXi log_phammer_bn_sign(b_size);
159  Eigen::VectorXi log_phammer_ap1mpn_sign(a_size);
160  Eigen::VectorXi log_phammer_bp1mpn_sign(b_size);
161  Eigen::VectorXi log_phammer_ap1m_sign = Eigen::VectorXi::Ones(a_size);
162  Eigen::VectorXi log_phammer_bp1m_sign = Eigen::VectorXi::Ones(b_size);
163 
164  // If the inner loop converges in 1 iteration, then the sum has coverged
165  // and another iteration of the outer loop is not needed
166  while ((n_iter > 1) && (m < outer_steps)) {
167  ap1n = ap1;
168  bp1n = bp1;
169  an = a_ref;
170  bn = b_ref;
171  ap1mn = ap1m;
172  bp1mn = bp1m;
173 
174  int n = 0;
175  Tz log_z_mn = log_z_m;
176  int z_pow_mn_sign = z_pow_m_sign;
177  scalar_t inner_diff = 0;
178  lgamma_np1 = 0;
179 
180  log_phammer_1n = 0;
181  log_phammer_an.setZero();
182  log_phammer_ap1_n.setZero();
183  log_phammer_bp1_n.setZero();
184  log_phammer_bn.setZero();
185  log_phammer_ap1_mpn = log_phammer_ap1_m;
186  log_phammer_bp1_mpn = log_phammer_bp1_m;
187  log_phammer_2_mpn = log_phammer_2m;
188  log_phammer_an_sign.setOnes();
189  log_phammer_ap1n_sign.setOnes();
190  log_phammer_bp1n_sign.setOnes();
191  log_phammer_bn_sign.setOnes();
192  log_phammer_ap1mpn_sign = log_phammer_ap1m_sign;
193  log_phammer_bp1mpn_sign = log_phammer_bp1m_sign;
194 
195  while ((inner_diff > log_precision) && (n < inner_steps)) {
196  // Numerator term
197  scalar_t term1_mn = log_z_mn + sum(log_phammer_ap1_mpn) + log_phammer_1m
198  + log_phammer_1n;
199  // Denominator term
200  scalar_t term2_mn = lgamma_mp1 + lgamma_np1 + sum(log_phammer_bp1_mpn)
201  + log_phammer_2_mpn;
202  int base_sign = z_pow_mn_sign * log_phammer_ap1mpn_sign.prod()
203  * log_phammer_bp1mpn_sign.prod();
204 
205  if (calc_a) {
206  // Division (on log scale) for the a & b partials
207  // Determine signs of each element
208  curr_signs_da = (base_sign * log_phammer_an_sign.array()
209  * log_phammer_ap1n_sign.array())
210  .matrix();
211  da_mn = (term1_mn + log_phammer_an.array())
212  - (term2_mn + log_phammer_ap1_n.array());
213 
214  // Aggregate the sums on the natural scale, so that the sign can be
215  // applied before aggregation
216  da += exp(da_mn).cwiseProduct(curr_signs_da);
217  }
218 
219  if (calc_b) {
220  curr_signs_db = (base_sign * log_phammer_bn_sign.array()
221  * log_phammer_bp1n_sign.array())
222  .matrix();
223  db_mn = (term1_mn + log_phammer_bn.array())
224  - (term2_mn + log_phammer_bp1_n.array());
225  db += exp(db_mn).cwiseProduct(curr_signs_db);
226  }
227 
228  // Series convergence assessed by whether the maximum term is
229  // smaller than the specified criteria (precision)
230  inner_diff = max(da_mn.maxCoeff(), db_mn.maxCoeff());
231 
232  // Increment the input arguments and rising factorials
233  log_z_mn += log_z;
234  log_phammer_1n += log1p(n);
235  log_phammer_2_mpn += log(2 + m + n);
236 
237  log_phammer_ap1_n += log(stan::math::fabs(ap1n));
238  log_phammer_bp1_n += log(stan::math::fabs(bp1n));
239  log_phammer_an += log(stan::math::fabs(an));
240  log_phammer_bn += log(stan::math::fabs(bn));
241  log_phammer_ap1_mpn += log(stan::math::fabs(ap1mn));
242  log_phammer_bp1_mpn += log(stan::math::fabs(bp1mn));
243 
244  z_pow_mn_sign *= z_sign;
245  log_phammer_ap1n_sign.array() *= sign(value_of_rec(ap1n)).array();
246  log_phammer_bp1n_sign.array() *= sign(value_of_rec(bp1n)).array();
247  log_phammer_an_sign.array() *= sign(value_of_rec(an)).array();
248  log_phammer_bn_sign.array() *= sign(value_of_rec(bn)).array();
249  log_phammer_ap1mpn_sign.array() *= sign(value_of_rec(ap1mn)).array();
250  log_phammer_bp1mpn_sign.array() *= sign(value_of_rec(bp1mn)).array();
251 
252  n += 1;
253  lgamma_np1 += log(n);
254  ap1n.array() += 1;
255  bp1n.array() += 1;
256  an.array() += 1;
257  bn.array() += 1;
258  ap1mn.array() += 1;
259  bp1mn.array() += 1;
260  }
261 
262  z_pow_m_sign *= z_sign;
263 
264  n_iter = n;
265 
266  log_z_m += log_z;
267  log_phammer_1m += log1p(m);
268  log_phammer_2m += log(2 + m);
269  log_phammer_ap1_m += log(stan::math::fabs(ap1m));
270  log_phammer_ap1m_sign.array() *= sign(value_of_rec(ap1m)).array();
271  log_phammer_bp1_m += log(stan::math::fabs(bp1m));
272  log_phammer_bp1m_sign.array() *= sign(value_of_rec(bp1m)).array();
273 
274  m += 1;
275 
276  lgamma_mp1 += log(m);
277  ap1m.array() += 1;
278  bp1m.array() += 1;
279  ap1mn.array() += 1;
280  bp1mn.array() += 1;
281  }
282 
283  if (m == outer_steps) {
284  throw_domain_error("grad_pFq", "k (internal counter)", outer_steps,
285  "exceeded ",
286  " iterations, hypergeometric function gradient "
287  "did not converge.");
288  }
289 
290  if (calc_a) {
291  auto pre_mult_a = (z * a_prod / a_ref.array() / b_prod).matrix();
292  std::get<0>(grad_tuple) = std::move(pre_mult_a.cwiseProduct(da));
293  }
294 
295  if (calc_b) {
296  auto pre_mult_b = ((z * a_prod) / (b.array() * b_prod)).matrix();
297  std::get<1>(grad_tuple) = std::move(-pre_mult_b.cwiseProduct(db));
298  }
299  }
300 
301  if (calc_z) {
302  std::get<2>(grad_tuple)
303  = std::move((a_prod / b_prod) * hypergeometric_pFq(ap1, bp1, z));
304  }
305 }

326 auto grad_pFq(const Ta& a, const Tb& b, const Tz& z, double precision = 1e-10,
327  int outer_steps = 1e6, int inner_steps = 1e6) {
328  using partials_t = partials_return_t<Ta, Tb, Tz>;
329  std::tuple<promote_scalar_t<partials_t, plain_type_t<Ta>>,
330  promote_scalar_t<partials_t, plain_type_t<Tb>>,
331  promote_scalar_t<partials_t, plain_type_t<Tz>>>
332  ret_tuple;
333  internal::grad_pFq_impl<!is_constant<Ta>::value, !is_constant<Tb>::value,
334  !is_constant<Tz>::value>(
335  ret_tuple, value_of(a), value_of(b), value_of(z), precision, outer_steps,
336  inner_steps);
337  return ret_tuple;
338 }
